FOR 

   range()	índice	quando precisa manipular posição
   
   enumerate()	índice + valor	quando precisa dos dois
      nomes = ["Pedro", "Ana", "Carlos"]

      for i, nome in enumerate(nomes):
          print(f"Posição {i}: {nome}")
   
   zip()	valor + valor	quando quer combinar listas

      nomes = ["Pedro", "Ana", "Carlos"]
      idades = [25, 30, 22]

      for nome, idade in zip(nomes, idades):
         print(f"{nome} tem {idade} anos")
